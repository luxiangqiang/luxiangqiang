微信小程序的生命周期！

<!--more-->

#### 一、小程序的生命周期

```javascript
//app.js
//功能：注册小程序（小程序的生命周期）
App({
  // 小程序的启动，全局只调用一次
 onLaunch:function(e){
  console.log(e+"初始化成功")
 },
 // 监听小程序页面显示——从后台进入前台
 onShow:function(){
   console.log("显示")
 },
 // 监听小程序隐藏——从前台进入后台
 onHide:function(){
   console.log("隐藏")
 },
 // 当小程序发生脚本错误时调用
 onError: function (msg){
   console.log(“报错”);
 }
})
```

#### 二、页面的生命周期

```javascript
Page({
  // 生命周期函数--监听页面加载
  onLoad: function (options) {
    console.log("page ---onLoad---");
  },
  // 生命周期函数--监听页面初次渲染完成
  onReady: function () {
    console.log("page ---onReady---");
  },
  // 生命周期函数--监听页面显示——后台进入前台
  onShow: function () {
    console.log("page ---onShow---");
  },
  // 生命周期函数--监听页面隐藏——前台进入后台
  onHide: function () {
    console.log("page ---onHide---");
  },
  // 生命周期函数--监听页面卸载——页面销毁
  onUnload: function () {
    console.log("page ---onUnload---");
  }
})
```
